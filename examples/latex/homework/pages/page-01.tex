\textbox[x=13em, y=39em, w=36em, h=10em]{
\[
\lim_{x \to 2} \frac{x^{2}-4}{x-2}
= \lim_{x \to 2} \frac{(x-2)(x+2)}{x-2}
= \lim_{x \to 2} (x+2)
= \boxed{4}
\]
}

\textbox[x=13em, y=57em, w=36em, h=10em]{
\[
f(x)=x^{3}\ln(x)-\frac{4}{x^{2}} = x^{3}\ln(x)-4x^{-2}
\]
\[
f'(x)=\left(x^{3}\ln(x)\right)'-4\left(x^{-2}\right)'
=3x^{2}\ln(x)+x^{3}\cdot \frac{1}{x}+8x^{-3}
=\boxed{3x^{2}\ln(x)+x^{2}+\frac{8}{x^{3}}}
\]
}
