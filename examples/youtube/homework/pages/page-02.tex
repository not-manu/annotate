\textbox[x=13em, y=20em, w=36em, h=20em]{
  \[
    \int_0^2 (3x^2 - 4x + 1)\,dx
    = \Bigl[x^3 - 2x^2 + x\Bigr]_0^2
    = (8 - 8 + 2) - 0
    = \boxed{2}
  \]
  \begin{center}
  \begin{tikzpicture}[scale=1.1, color=annotate]
    % shaded area between curve and x-axis on [0,2]
    \fill[blue!15]
      plot[domain=0:0.3333,samples=60] (\x, {3*\x*\x - 4*\x + 1}) -- (0.3333,0) -- (0,0) -- cycle;
    \fill[blue!15]
      (0.3333,0) -- plot[domain=0.3333:1,samples=60] (\x, {3*\x*\x - 4*\x + 1}) -- (1,0) -- cycle;
    \fill[blue!15]
      (1,0) -- plot[domain=1:2,samples=60] (\x, {3*\x*\x - 4*\x + 1}) -- (2,0) -- cycle;
    % axes
    \draw[->] (-0.2,0) -- (2.5,0) node[right] {$x$};
    \draw[->] (0,-1.2) -- (0,5.5) node[above] {$y$};
    % curve
    \draw[thick] plot[domain=-0.05:2.15,samples=100] (\x, {3*\x*\x - 4*\x + 1});
    % tick marks and labels
    \foreach \xv in {1,2} {
      \draw (\xv,0.05) -- (\xv,-0.05) node[below] {\small$\xv$};
    }
    \foreach \yv in {1,2,3,4,5} {
      \draw (0.05,\yv) -- (-0.05,\yv) node[left] {\small$\yv$};
    }
    \draw (0.05,-1) -- (-0.05,-1) node[left] {\small$-1$};
    % roots
    \filldraw (0.3333,0) circle (1.2pt);
    \filldraw (1,0) circle (1.2pt);
    % integration bounds
    \draw[dashed] (0,0) -- (0,1);
    \draw[dashed] (2,0) -- (2,5);
    % label
    \node at (1.5, 1.5) {\small$y=3x^2-4x+1$};
  \end{tikzpicture}
  \end{center}
}
